package com;

public class SalaryCalculation {
	/** 
	 * Nếu số ngày nghỉ của nhân viên  <=7 thì nhân viên sẽ bị trừ lương tháng. Số tiền trừ  = số ngày nghỉ * (tiền lương/22) * 1.0
	 * Nếu số ngày nghỉ là >7 và <= 11 thì nhân viên sẽ bị trừ lương. Số tiền trừ  = số ngày nghỉ * (tiền lương/22) * 1.2
	 * Nếu số ngày nghỉ là >11 thì nhân viên sẽ bị trừ lương. Số tiền trừ  = số ngày nghỉ * (tiền lương/22) * 1.3
	 * Nếu nhân viên làm thêm giờ thì số tiền lương làm thêm giờ  = (tiền lương/22)/2 * số ngày làm thêm giờ
	 * @param dayOff là ngày nghỉ trong tháng >=0 và <=22
	 * @param overTime là số ngày làm thêm giờ trong tháng phải >=0 và <=22
	 * @param salary là lương tháng của nhân viên phải >=1000 và <=100000
	 * @return số tiền lương của nhân viên hoặc -1 nếu tham số không hợp lệ.
	 */
	 public double calculate(int dayOff, int overTime, double salary) {
		 if (dayOff < 0 || dayOff > 22
			|| overTime < 0 || overTime > 22
			|| salary < 1000 || salary > 100000) {
				return -1;
		}
		double salaryPerday  = salary/22;
		double overTimeSalary = salaryPerday * overTime;
		double penalty = 0;
		if (dayOff <= 7) {
			penalty = salaryPerday * dayOff;
		} else if (dayOff <=11) (
			penalty = salaryPerday * dayOff * 1.2;
		} else {
			penalty = salaryPerday * dayOff * 1.3;
		}
	 
		return salary - penalty + overTimeSalary;
	 }

}
